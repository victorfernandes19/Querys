SELECT

CONTREV.NU_PROTOCOLO_PR            AS PROTOCOLO_REV, 
CONTREV.DT_PEDIDO_PROTOCOLO_PR     AS DATA_PEDIDO,
CONTREV.DT_PAGTO_ACERTO_PR        AS DATA_PAGTO, 
CONTREV.DS_TIPO_PROTOCOLO_PR       AS TIPO_PROT,
CONTREV.DS_SITUACAO_RECURSO_PR     AS SITUACAO_PROT,
CONTPRE.CD_PRESTADOR		           AS COD_PREST,
CONTPRE.DS_VIP			               AS VIP,
CONTREV.DS_ORIGEM_PROTOCOLO_PR     AS ORIGEM,		
--CONTPRE.NO_PRESTADOR		   AS NOME_PRES,
CONTREV.NU_GUIA_GA		             AS GUIA,	
CONTREV.DS_FASE_GUIA_RG            AS FASE_GUIA,
sum(CONTREV.QTD_RECURSO_RE)        AS QTD_RECURSADA,
CONTREV.VLR_APRESENTADO_RE         AS VLR_RECURSADO,
CONTREV.VLR_ACERTO_RE		           AS VLR_ACERTO,
CONTREV.VLR_PASSIVEL_COMPL_RECURSO_RE AS VLR_PASSIVEL_RECURSO,
CONTREV.DT_ATENDIMENTO_RE          AS DATA_ATENDIMENTO,
CONTREV.NU_EVENTO_IMPORTADO_RE     AS ESTRUTURA_REV,
CONTREV.DS_EVENTO_IMPORTADO_RE     AS DESCRICAO_EVENTO_REV,
CONTREV.DS_SITUACAO_EV_RE          AS SITUACAO_EVENTO,
CONTREV.DS_TERMO_RE                AS TERMO_RECURSADO,
CONTREV.DS_MOTIVO_DEFERIMENTO_RE  AS MOT_DEF,
CONTREV.DS_MOTIVO_INDEFERIMENTO_RE AS MOT_INDEF,
CONTREV.NU_GUIA_GA 		   AS GUIA,
CONTREV.DS_MODELO_GUIA_GA	   AS TIPO_GUIA,
CONTREV.DS_UF_PEG_PA               AS UF


FROM 
gold.apcm_deacm_contasmedicas.tb_contas_medicas_rev   CONTREV
join gold.apcm_deacm_contasmedicas.tb_contas_medicas_prestador  CONTPRE   on  CONTPRE.HD_PRESTADOR = CONTREV.HD_PRESTADOR_PR


WHERE

CONTPRE.DS_VIP = 'S'
AND CONTREV.DS_SITUACAO_RECURSO_PR <> "Em an√°lise"
AND CONTREV.DT_PAGTO_ACERTO_PR BETWEEN '2024-10-01' AND current_date()
AND CONTREV.DS_ORIGEM_PROTOCOLO_PR = "Web"



GROUP BY


CONTREV.NU_PROTOCOLO_PR, 
CONTREV.DT_PEDIDO_PROTOCOLO_PR,
CONTREV.DT_PAGTO_ACERTO_PR,
CONTREV.DS_TIPO_PROTOCOLO_PR ,
CONTREV.DS_SITUACAO_RECURSO_PR ,
CONTPRE.CD_PRESTADOR,
CONTPRE.DS_VIP,			
--CONTPRE.NO_PRESTADOR		   ,
CONTREV.NU_GUIA_GA,	
CONTREV.DS_FASE_GUIA_RG,
CONTREV.QTD_RECURSO_RE        ,
CONTREV.VLR_APRESENTADO_RE,
CONTREV.VLR_ACERTO_RE,
CONTREV.DT_ATENDIMENTO_RE,
CONTREV.NU_EVENTO_IMPORTADO_RE,
CONTREV.DS_EVENTO_IMPORTADO_RE,
CONTREV.DS_SITUACAO_EV_RE,
CONTREV.DS_TERMO_RE,
CONTREV.VLR_PASSIVEL_COMPL_RECURSO_RE,
CONTREV.DS_MOTIVO_DEFERIMENTO_RE,
CONTREV.DS_MOTIVO_INDEFERIMENTO_RE,
CONTREV.DS_ORIGEM_PROTOCOLO_PR,
CONTREV.NU_GUIA_GA, 		
CONTREV.DS_MODELO_GUIA_GA,
CONTREV.DS_UF_PEG_PA
